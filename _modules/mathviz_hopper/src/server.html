    def __init__(self, obj):
        app = Bottle()
        self._setup_routing(app)
        self.server = MyWSGIRefServer(host="localhost", port=obj.port)
        self.index = obj.index
        self.settings = obj.settings
        def begin():
            run(app, server=self.server)
        Thread(target=begin).start()

    def __delete__(self):
        self.server.stop()
